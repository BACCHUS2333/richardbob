#random forest
def create_random_forest(X_train, y_train, n_estimators=100,max_depth = None,max_features='sqrt',random_state=69):
    """
    Create a random forest classifier using sklearn.
    
    Parameters:
        X_train (array-like): The training input samples.
        y_train (array-like): The target values.
        n_estimators (int): The number of trees in the forest (default=100).
        max_depth (int): The maximum depth of the tree (default=None).
    
    Returns:
        RandomForestClassifier: The random forest classifier model.
    """
    # Create the random forest classifier
    print("n_estimators: ",n_estimators)
    print("max_features: ",max_features)
    rf = RandomForestClassifier(n_estimators=n_estimators, max_depth=max_depth, random_state=random_state,max_features=max_features)
    
    # Fit the model to the training data
    rf.fit(X_train, y_train)
    
    # Return the trained model
    return rf

#apply random forest on multiple n estimators
confusion_matrix_list = [] 
accuracy_score_list = []
precision_list = []
recall_list = []
f1_score_list = []


def rf_tree_number(tree_number,max_features='sqrt'):
    X = std_data.drop('y', axis=1)
    y = std_data['y']
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.33, random_state=42)
    rf = create_random_forest(X_train, y_train,n_estimators=tree_number,max_features=max_features)
    X_predicted_rf = rf.predict(X_test)
    print("Accuracy:\n ", accuracy_score(y_test, X_predicted_rf))
    print("Confusion Matrix: \n", confusion_matrix(y_test, X_predicted_rf))
    print("Classification Report:\n ", classification_report(y_test, X_predicted_rf))
    confusion_matrix_list.append(confusion_matrix(y_test, X_predicted_rf))
    accuracy_score_list.append(accuracy_score(y_test, X_predicted_rf))
    precision_list.append(classification_report(y_test, X_predicted_rf,output_dict=True)['1']['precision'])
    recall_list.append(classification_report(y_test, X_predicted_rf,output_dict=True)['1']['recall'])
    f1_score_list.append(classification_report(y_test, X_predicted_rf,output_dict=True)['1']['f1-score'])

    return rf

Sometimes, performance metrics exhibit an alternating pattern when the number of trees in a random forest model is very small. This is because with fewer trees, the predictions of individual trees can be more variable due to the random nature of the algorithm, in other words, due to the features included in the decision trees. A limited number of decision trees may lead to a partial inclusion of all features in the training sample, rendering a different pattern of the result.
