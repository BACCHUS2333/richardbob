#smote_nc

def smote_nc(X_train_data,y_train_data, ratio, k_neighbors):

    smote_nc = SMOTENC(categorical_features=[3,4,5,6], random_state=123,sampling_strategy=ratio, k_neighbors=k_neighbors)
    X_train_resampled, y_train_resampled = smote_nc.fit_resample(X_train_data, y_train_data)
    return X_train_resampled, y_train_resampled

def dt(X_train, y_train, X_test, y_test):
    dt = DecisionTreeClassifier(random_state=123)
    dt.fit(X_train, y_train)
    
    cfm_train = confusion_matrix(y_train, dt.predict(X_train))
    TP_train = cfm_train[1,1]
    TN_train = cfm_train[0,0]
    FP_train = cfm_train[0,1]
    FN_train = cfm_train[1,0]
    accuracy_train = (TP_train+TN_train)/(TP_train+TN_train+FP_train+FN_train)
    precision_train = TP_train/(TP_train+FP_train)
    recall_train = TP_train/(TP_train+FN_train)
    f1_train = 2*precision_train*recall_train/(precision_train+recall_train)

    y_pred = dt.predict(X_test)
    cfm = confusion_matrix(y_test, y_pred)
    TP = cfm[1,1]
    TN = cfm[0,0]
    FP = cfm[0,1]
    FN = cfm[1,0]
    accuracy = round((TP+TN)/(TP+TN+FP+FN),4)
    precision = round(TP/(TP+FP),4)
    recall = round(TP/(TP+FN),4)
    f1 = round(2*precision*recall/(precision+recall),4)
    return accuracy_train, precision_train, recall_train, f1_train,accuracy, precision, recall, f1
