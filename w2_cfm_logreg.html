def threshold_confusion_table(predicted_value,threshold,actual_value,print =False  ):
        
        labels = np.array([0]*len(predicted_value))
        labels[predicted_value >threshold] = 1
        if print: display(confusion_table(labels , actual_value))
        confusion_table1 = confusion_table(labels , actual_value)
        #confusion_table_image = pd.DataFrame(data= [['tn','fn'],['fp','tp']],columns = ['true 0','true 1'],index = ['predicted 0','predicted 1'])
        if print: display("confusion matrix image:",
                confusion_table_image)
        #display('correct rate:',(confusion_table1['tp']+confusion_table1['tn'])/len(predicted_value))
        return confusion_table1,confusion_table_image


        
def random_categorizer(actual_value,threshold,print =False):
    predicted_value = np.random.binomial(1,0.5,len(actual_value))
    
    tp = np.sum(np.logical_and(actual_value == 1, predicted_value == 1))
    tn = np.sum(np.logical_and(actual_value == 0, predicted_value == 0))
    fp = np.sum(np.logical_and(actual_value == 0, predicted_value == 1))
    fn = np.sum(np.logical_and(actual_value == 1, predicted_value == 0))
    random_confusion_table1 = pd.DataFrame(data = [[tn,fn],[fp,tp]],columns = ['true 0','true 1'],index = ['predicted 0','predicted 1'])
    if print: display(random_confusion_table1)
    return random_confusion_table1,confusion_table_image

#write a function to calculate the accuray, precision, recall, and F1 score
def calculate_metrics(tp, fp, tn, fn):
    accuracy = (tp + tn) / (tp + fp + tn + fn)
    precision = tp / (tp + fp)
    recall = tp / (tp + fn)
    f1_score = 2 * (precision * recall) / (precision + recall)
    
    return pd.DataFrame(data = [accuracy, precision, recall, f1_score], index = ['accuracy', 'precision', 'recall', 'f1_score'], columns = ['value'])
