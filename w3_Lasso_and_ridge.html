def lasso_regression(X, y):
    lasso = Lasso()
    count = 0
    if alphas_lasso:
        alphas = np.linspace(0.01, 60, 1000)
    else:
        alphas  = [0.01, 0.1, 1, 10, ]
    scores_across_alpha = pd.DataFrame(columns=['alpha', 'mean_score','median_score','max_score','min_score'])
    
    for alpha in alphas:
        lasso.alpha = alpha
        scores = cross_val_score(lasso, X, y, cv=20, scoring='neg_mean_squared_error')
        mean_score = -np.mean(scores)
        median_score = -np.median(scores)
        max_score = -np.max(scores)
        min_score = -np.min(scores)
        
        #scores_across_alpha[len(scores_across_alpha)] = [alpha, mean_score,median_score,max_score,min_score]
        
        scores_across_alpha = pd.concat([scores_across_alpha, pd.DataFrame({'alpha':alpha, 'mean_score':mean_score,'median_score':median_score,'max_score':max_score,'min_score':min_score}, index=[0])], ignore_index=True)
        #scores_across_alpha = scores_across_alpha.append({'alpha':alpha, 'mean_score':mean_score,'median_score':median_score,'max_score':max_score,'min_score':min_score}, ignore_index=True)
        count += 1
        if count % 100 == 0:
            print("{} times iterations".format(count))
    return scores_across_alpha

from sklearn.linear_model import RidgeCV
from sklearn.model_selection import LeaveOneOut
import warnings
warnings.filterwarnings("ignore")
#alphas_ridge = True
def ridge_regression(X, y,x_test,y_test):
    ridge = RidgeCV(cv=LeaveOneOut())
    count = 0
    if alphas_ridge:
        alphas = np.linspace(0.01, 60, 1000)
    else:
        alphas = [0.01,0.1,1,10]
    scores_across_alpha_ridge = pd.DataFrame(columns=['alpha', 'mean_score','test_error','coefficients'])
    for alpha in alphas:
        ridge.alpha_ = alpha
        ridge.fit(X, y)
        score = ridge.score(X, y)
        test_error_ridge = np.mean((ridge.predict(x_test)-y_test)**2)
        scores_across_alpha_ridge = pd.concat([scores_across_alpha_ridge, pd.DataFrame
        ({'alpha':alpha, 'mean_score':score,'test_error': test_error_ridge,'coefficients':ridge.coef_[0]}, index=[0])], ignore_index=True)

        count += 1
        if count % 20 == 0:
            print("{} times iterations".format(count))

        #scores = cross_val_score(ridge, X, y, cv=20, scoring='neg_mean_squared_error')
        
    return scores_across_alpha_ridge
